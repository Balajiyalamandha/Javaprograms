pipeline {
    agent any
    
    environment {
        // Optional: Set Java version if multiple installed
        JAVA_HOME = 'C:\\Program Files\\Java\\jdk-17'
        PATH = "${env.JAVA_HOME}\\bin;${env.PATH}"
    }

    stages {
        stage('Checkout') {
            steps {
                echo 'Cloning repository...'
                git branch: 'main', url: 'https://github.com/Balajiyalamandha/Javaprograms.git'
            }
        }

        stage('Build') {
            steps {
                echo 'Building Java project...'
                // For Maven
                bat 'mvn clean compile'
                // For Gradle: bat 'gradle build'
            }
        }

        stage('Run Tests') {
            steps {
                echo 'Running tests...'
                // For Maven
                bat 'mvn test'
                // For Gradle: bat 'gradle test'
            }
            post {
                always {
                    junit '**/target/surefire-reports/*.xml' // Test report publishing
                }
            }
        }

        stage('Package') {
            steps {
                echo 'Packaging artifact...'
                bat 'mvn package'
            }
        }
    }

    post {
        success {
            echo 'Build and tests SUCCESSFUL!'
            // Optional: Notify Slack/email here
        }
        failure {
            echo 'Build or tests FAILED!'
            // Optional: Notify Slack/email here
        }
        always {
            echo 'Cleaning workspace...'
            cleanWs()
        }
    }
}
