pipeline {
    agent any

    tools {
        jdk 'JDK 17'           // Matches the JDK configured in Jenkins
        maven 'Maven 3.8'      // Matches Maven configured in Jenkins
    }

    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/Balajiyalamandha/Javaprograms.git'
            }
        }
        stage('Build') {
            steps {
                sh 'mvn clean compile'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Package') {
            steps {
                sh 'mvn package'
            }
        }
    }
}
